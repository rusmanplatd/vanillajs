üìù Component Class (TypeScript-like)

class CounterComponent extends ReactiveComponent {
  // @Input decorators
  static _inputs = {
    'initial-count': 'initialCount',
    'max-count': 'maxCount'
  };

  // @Output decorators
  static _outputs = ['countChanged'];

  constructor() {
    super();
    this.count = 0;
  }

  ngOnInit() {
    this.count = parseInt(this.initialCount, 10) || 0;
  }

  increment() {
    if (this.count < this.maxCount) {
      this.count++;
      this.countChanged.emit(this.count);
      this.detectChanges();
    }
  }

  decrement() {
    this.count--;
    this.countChanged.emit(this.count);
    this.detectChanges();
  }
}

Component({
  selector: 'counter-component',
  templateUrl: './counter.component.html',
  styleUrl: './counter.component.css'
})(CounterComponent);

<div class="counter-container">
  <h2>Counter Component</h2>

  </div>
  <!-- Structural directives -->
  <div *ngIf="count > 10">Count is high!</div>
  <div *ngIf="count < 0">Count is negative!</div>
</div>
